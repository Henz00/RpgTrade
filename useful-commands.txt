postgresql terminal:
docker exec -it rpgtrade-postgres psql -U postgres -d rpgtrade

docker:
docker compose up -d
docker compose down

docker compose -f docker-compose.prod.yml down
docker compose -f docker-compose.prod.yml up -d --build

status check:
docker compose -f docker-compose.prod.yml ps

create migrations:
dotnet ef migrations add InitialCreate --project RpgTrade.Infrastructure --startup-project RpgTrade.Api --output-dir Persistence/Migrations

update the database schema from migrations:
dotnet ef database update --project RpgTrade.Infrastructure --startup-project RpgTrade.Api

start API:
dotnet run --project RpgTrade.Api
